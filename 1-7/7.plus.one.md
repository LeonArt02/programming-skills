# ðŸ’™ Code
```python
class Solution(object):
    def plusOne(self, digits):
        """
        :type digits: List[int]
        :rtype: List[int]
        """
        dig_str = [str(i) for i in digits]
        new_num = int(''.join(dig_str)) + 1
        new_dig_list = [int(i) for i in str(new_num)]
        return new_dig_list
```
