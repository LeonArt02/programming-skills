class Solution(object):
    def romanToInt(self, s):
        rom = {'I':1, 'V':5, 'X':10, 'L':50,'C':100, 'D':500, 'M':1000}
        re = 0
        for i in range(len(s)):
            if i + 1 < len(s) and rom[s[i]] < rom[s[i + 1]]:
                re -= rom[s[i]]
            else:
                re += rom[s[i]]
        return re
