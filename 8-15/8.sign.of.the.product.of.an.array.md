# ðŸ’™ Code
```python
class Solution(object):
    def arraySign(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        product = nums.pop(0)
        for n in nums:
            product *= n
        if product > 0:
            return 1
        elif product < 0:
            return -1
        else:
            return 0
```
